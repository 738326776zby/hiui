TimePicker 时间选择器

基础

时间选择

时间范围选择

至
import React from 'react'
import TimePicker from '@hi-ui/hiui/es/time-picker/TimePicker'
class Demo extends React.Component {
  render() {
    return (
      <div style={{display:'flex', flexWrap: 'wrap'}}>
        <div style={{margin: '10px'}}>
          <p>时间选择</p>
          <TimePicker
            value={new Date()}
            format="HH:mm"
            onChange={date => console.log('时间', date)}
          />
        </div>
        <div style={{margin: '10px'}}>
          <p>时间范围选择</p>
          <TimePicker
            value={new Date()}
            type="timerange"
            format="HH:mm"
            onChange={date => console.log('时间范围', date)}
          />
        </div>
      </div>
    )
  }
}
