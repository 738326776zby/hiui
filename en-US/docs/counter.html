Counter 数字加减器

基础

-+
-+
import React from 'react'
import Counter from '@hiui/hiui/es/counter'
class Demo extends React.Component {
  constructor() {
    super()
    this.state = {
      disabled: false
    }
  }
  render() {
    return (
      <div>
        <Counter
          value='4'
          step='1'
          min='-10'
          max='10'
          onChange={(val) => console.log('变化后的值：', val)}
        />
        <Counter
          value='8888'
          step='2'
          min='2'
          max='8'
          disabled={this.state.disabled}
          onChange={(e, val) => {console.log('----val', e, val)}}
        />
      </div>
    )
  }
}
